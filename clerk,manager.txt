CREATE DATABASE EMPLOYEES;
USE EMPLOYEES;
USE EMPLOYEES;
CREATE TABLE EMPLOYEE(ENO INT(4),ENAME VARCHAR(30),EPNO INT(10));
CREATE USER 'clerks' identified by 'clerk1';
CREATE USER 'managers' identified by 'man1';
GRANT UPDATE(EPNO) ON EMPLOYEES.EMPLOYEE TO clerks;
GRANT SELECT,DELETE,UPDATE,INSERT ON EMPLOYEES.EMPLOYEE TO managers;
SHOW GRANTS FOR clerks;
SHOW GRANTS FOR managers;
REVOKE ALTER ON EMPLOYEES.EMPLOYEE TO clerks;
SHOW GRANTS FOR clerks;
GRANT INSERT ON EMPLOYEES.EMPLOYEE TO clerks;
SHOW GRANTS FOR clerks;
